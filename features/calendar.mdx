## Overview

ClubConnect UK uses **react-big-calendar** with custom extensions for drag-and-drop, multiple views, and color-coded events. This document covers the technical implementation for developers working with the calendar system.

The calendar is the core visualization for the [Cover Scheduling](/features/cover-scheduling) feature.

## Data Transformation

Cover occurrences must be transformed into calendar events:

```typescript
const calendarEvents = useMemo((): Array<CalendarEvent> => {
  return events.map((occ) => {
    const meetingDate = parseLocalDate(occ.meeting_date)

    let start = new Date(meetingDate)
    let end = new Date(meetingDate)
    let displayEnd = new Date(meetingDate)

    if (occ.cover_rule) {
      const [startH, startM = 0] = occ.cover_rule.start_time
        .split(':')
        .map(Number)
      const [endH, endM = 0] = occ.cover_rule.end_time.split(':').map(Number)

      start = set(meetingDate, {
        hours: startH,
        minutes: startM,
        seconds: 0,
        milliseconds: 0,
      })
      
      displayEnd = set(meetingDate, {
        hours: endH,
        minutes: endM,
        seconds: 0,
        milliseconds: 0,
      })
      
      // react-big-calendar requires end to be +1 hour for correct rendering
      end = addHours(displayEnd, 1)
    } else {
      // Default 9-10 AM if no rule
      start = set(meetingDate, { hours: 9, minutes: 0, seconds: 0 })
      displayEnd = set(meetingDate, { hours: 10, minutes: 0, seconds: 0 })
      end = addHours(displayEnd, 1)
    }

    return {
      id: occ.id,
      title:
        occ.cover_rule?.club?.club_name ||
        occ.cover_rule?.school?.school_name ||
        'Cover',
      start,
      end,
      displayStart: start,
      displayEnd,
      resource: occ,  // Store full occurrence for access in handlers
    }
  })
}, [events])
```

<Note>
  The `end` time is 1 hour past `displayEnd` due to react-big-calendar's rendering logic. Use `displayEnd` for display purposes.
</Note>

## Color System

School-based color generation ensures consistent visual identification:

```typescript
export const getSchoolColors = (schoolName: string) => {
  // Hash school name to generate consistent color
  const hash = schoolName.split('').reduce((acc, char) => {
    return char.charCodeAt(0) + ((acc << 5) - acc)
  }, 0)

  // Generate HSL color
  const hue = Math.abs(hash % 360)
  const saturation = 65 + (Math.abs(hash % 20))
  const lightness = 88  // Light background

  // Convert to hex for styling
  const bgHex = hslToHex(hue, saturation, lightness)
  const textHex = hslToHex(hue, saturation, 25)  // Dark text
  const borderHex = hslToHex(hue, saturation, 50)  // Medium border

  return { bgHex, textHex, borderHex }
}
```

**Color Palette Example:**
- Riverside Academy: Light green background, dark green text
- Oak School: Light blue background, dark blue text
- Westside Primary: Light orange background, dark orange text

```typescript
const handleEventDrop = (info: EventDropInfo) => {

// In parent component:
const handleEventDrop = async (
  occurrence: CoverOccurrence,
  newDate: Date,
) => {
  const dateStr = format(newDate, 'yyyy-MM-dd')

  try {
    await moveOccurrence.mutateAsync({
      id: occurrence.id,
      newDate: dateStr,
    })
    toast.success(`Moved cover to ${format(newDate, 'PPP')}`)
  } catch (error) {
    console.error('Failed to move cover', error)
    toast.error('Failed to move cover session')
  }
}
```

### Drop from Outside

Supports dragging events from popovers to calendar:

```typescript
const [draggedOutsideEvent, setDraggedOutsideEvent] =
  useState<CalendarEvent | null>(null)

// In popover event list:
<div
  draggable
  onDragStart={() => setDraggedOutsideEvent(event)}
  onClick={(e) => {
    e.stopPropagation()
    onSelectEvent(event.resource)
  }}
>
  <MonthEventComponent event={event} />
</div>

// In calendar props:
onDropFromOutside={(info: DropFromOutsideInfo) => {
  if (draggedOutsideEvent) {
    handleEventDrop({ 
      event: draggedOutsideEvent, 
      start: info.start 
    })
    setDraggedOutsideEvent(null)
  }
}}
```

## View Configuration

### Calendar Props

```typescript
<DragAndDropCalendar
  localizer={localizer}
  events={calendarEvents}
  startAccessor="start"
  endAccessor="end"
  style={{ height: '100%' }}
  toolbar={false}  // Custom toolbar in parent
  view={view}
  onView={handleViewChange}
  date={date}
  onNavigate={handleNavigate}
  defaultView={Views.WEEK}
  views={[Views.MONTH, Views.WEEK, Views.WORK_WEEK, Views.DAY]}
  step={30}  // 30-minute intervals
  showMultiDayTimes
  min={set(new Date(), { hours: 7, minutes: 0 })}   // 7 AM
  max={set(new Date(), { hours: 22, minutes: 0 })}  // 10 PM
  scrollToTime={set(new Date(), { hours: 9, minutes: 0 })}  // Auto-scroll to 9 AM
  draggableAccessor={() => true}
  selectable={true}
  onSelectSlot={(slotInfo: SlotInfo) => {
    // Prevent clicking gray (off-month) days in month view
    if (view === Views.MONTH) {
      if (slotInfo.start.getMonth() !== date.getMonth()) return
    }
    onSelectSlot?.(slotInfo.start)
  }}
  onDrillDown={(drillDate: Date) => {
    if (view === Views.MONTH) {
      if (drillDate.getMonth() !== date.getMonth()) return
    }
    onDrillDown?.(drillDate)
  }}
  onEventDrop={handleEventDrop}
  onDropFromOutside={handleDropFromOutside}
  components={{
    event: view === Views.MONTH ? MonthEventComponent : WeekEventComponent,
    dateCellWrapper: CustomDateCellWrapper,
    showMore: CustomShowMore,
  }}
  eventPropGetter={() => ({
    className: cn(
      'cursor-grab active:cursor-grabbing hover:opacity-90 transition-opacity',
      view === Views.MONTH ? 'month-event' : 'week-event',
    ),
    style: {
      border: 'none',
    },
  })}
  onSelectEvent={(event: CalendarEvent) => onSelectEvent(event.resource)}
  formats={{
    dayHeaderFormat: (d: Date) => format(d, 'EEE d'),
    timeGutterFormat: (d: Date) => format(d, 'h a'),
    eventTimeRangeFormat: () => '',  // Hide default time (rendered in component)
  }}
/>
```

## Performance Optimizations

<AccordionGroup>
  <Accordion title="Event Memoization">
    Use `useMemo` for event transformation to avoid recalculating on every render:
    
    ```typescript
    const calendarEvents = useMemo(() => {
    ```

  </Accordion>

  <Accordion title="Lazy Loading">

    ```typescript
    const { data: occurrences } = useCoverOccurrences({

      endDate: format(endOfMonth(date), 'yyyy-MM-dd')
    ```

  </Accordion>

    ```typescript
    import { useVirtualizer } from '@tanstack/react-virtual'
    ```
