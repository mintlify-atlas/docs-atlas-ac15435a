## Overview

ClubConnect UK uses `@tanstack/react-query` for efficient data fetching, caching, and state management. All hooks are organized by feature domain.

## Installation

React Query is already configured in the project. To use hooks:

```typescript
import { useQuery, useMutation } from '@tanstack/react-query'
import { supabase } from '@/services/supabase'
```

## Schools

### useSchools

Fetches all active schools.

```typescript
import { useSchools } from '@/hooks/use-schools'

function SchoolsList() {
  const { data: schools, isLoading, error } = useSchools()
  
  if (isLoading) return <div>Loading...</div>
  if (error) return <div>Error: {error.message}</div>
  
  return (
    <ul>
      {schools?.map(school => (
        <li key={school.id}>{school.school_name}</li>
      ))}
    </ul>
  )
}
```

<ResponseField name="data" type="Array<School>">
  Array of active schools ordered by name
</ResponseField>

<ResponseField name="isLoading" type="boolean">
  True while fetching data
</ResponseField>

<ResponseField name="error" type="Error | null">
  Error object if query fails
</ResponseField>

**Query Key**: `['schools']`

**Filters**: Only returns schools with `status = 'active'`

**Sorting**: Ordered by `school_name` ascending

## Club Mutations

### useCreateClub

Creates a new club.

```typescript
import { useCreateClub } from '@/features/clubs/api/mutations'

function CreateClubForm() {
  const createClub = useCreateClub()
  
  const handleSubmit = async (formData: any) => {
    try {
      const newClub = await createClub.mutateAsync({
        club_name: formData.name,
        club_code: formData.code,
        school_id: formData.schoolId,
        description: formData.description,
        status: 'active'
      })
      console.log('Created:', newClub)
    } catch (error) {
      console.error('Failed to create club:', error)
    }
  }
  
  return <form onSubmit={handleSubmit}>...</form>
}
```

<ParamField path="club_name" type="string" required>
  Name of the club
</ParamField>

<ParamField path="club_code" type="string" required>
  Short identifier code
</ParamField>

<ParamField path="school_id" type="string" required>
  UUID of the school
</ParamField>

<ParamField path="description" type="string">
  Club description
</ParamField>

<ParamField path="status" type="ClubStatus">
  Initial status (default: 'active')
</ParamField>

<ResponseField name="data" type="Club">
  Created club object
</ResponseField>

**Invalidates**: `['clubs']`

### useUpdateClub

Updates an existing club.

```typescript
import { useUpdateClub } from '@/features/clubs/api/mutations'

function EditClubForm({ club }: { club: Club }) {
  const updateClub = useUpdateClub()
  
  const handleUpdate = async (updates: Partial<Club>) => {
    await updateClub.mutateAsync({
      id: club.id,
      ...updates
    })
  }
  
  return <form>...</form>
}
```

<ParamField path="id" type="string" required>
  UUID of the club to update
</ParamField>

<ParamField path="...updates" type="Partial<Club>">
  Fields to update (club_name, description, status, etc.)
</ParamField>

**Invalidates**: `['clubs']`, `['clubs', clubId]`

### useDeleteClub

```typescript
import { useDeleteClub } from '@/features/clubs/api/mutations'

function DeleteClubButton({ clubId }: { clubId: string }) {
```

## Cover Occurrences

### useCoverOccurrences

Fetches cover occurrences with flexible filtering.

```typescript
import { useCoverOccurrences } from '@/hooks/use-covers'

function Calendar() {
  const { data: occurrences } = useCoverOccurrences({
    schoolId: 'uuid-here',
    startDate: '2026-03-01',
    endDate: '2026-03-31'
  })
  
  return (
    <div>
      {occurrences?.map(occ => (
        <div key={occ.id}>
          {occ.cover_rule?.club?.club_name} on {occ.meeting_date}
        </div>
      ))}
    </div>
  )
}
```

<ParamField path="schoolId" type="string">
  Filter by school UUID (use 'all' for all schools)
</ParamField>

<ParamField path="startDate" type="string">
  ISO date string for range start
</ParamField>

<ParamField path="endDate" type="string">
  ISO date string for range end
</ParamField>

<ResponseField name="data" type="Array<CoverOccurrence>">
  Occurrences with nested cover_rule, club, school, and assignments
</ResponseField>

**Query Key**: `['cover_occurrences', { schoolId, startDate, endDate }]`

**Includes**: Full nested data for cover rules, clubs, schools, teachers, and person details

## Query Invalidation

After mutations, React Query automatically refetches related data. Common invalidation patterns:

```typescript
import { useQueryClient } from '@tanstack/react-query'

function MyComponent() {
  const queryClient = useQueryClient()
  
  // Invalidate all clubs queries
  queryClient.invalidateQueries({ queryKey: ['clubs'] })
  
  // Invalidate specific club
  queryClient.invalidateQueries({ queryKey: ['clubs', clubId] })
  
  // Invalidate multiple query keys
  queryClient.invalidateQueries({ queryKey: ['cover_occurrences'] })
  queryClient.invalidateQueries({ queryKey: ['teachers'] })
}
```
