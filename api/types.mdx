## Overview

ClubConnect UK uses TypeScript for type safety across the application. This page documents all interfaces and types used in the codebase, organized by domain.

## Core Types

### PersonDetails

Represents personal information for individuals in the system.

```typescript
interface PersonDetails {
  id: string
  first_name: string

  last_name: string

  address?: string | null
  contact?: string | null
  image?: string | null
  created_at: string
  updated_at: string

```

<ResponseField name="id" type="string" required>

</ResponseField>

<ResponseField name="first_name" type="string" required>

## Cover and Scheduling Types

### CoverRule

```typescript
type CoverFrequency = 'weekly' | 'bi-weekly' | 'monthly'
type DayOfWeek = 'monday' | 'tuesday' | 'wednesday' | 'thursday' | 'friday' | 'saturday' | 'sunday'
type CoverRuleStatus = 'active' | 'inactive' | 'cancelled'

interface CoverRule {
  id: string
  school_id: string
  club_id: string
  frequency: CoverFrequency
  day_of_week: DayOfWeek
  start_time: string
  end_time: string
  status: CoverRuleStatus
  created_at: string
  updated_at: string
  club?: Club
  school?: School
}
```

<ResponseField name="frequency" type="CoverFrequency" required>
  How often the session repeats
</ResponseField>

<ResponseField name="day_of_week" type="DayOfWeek" required>
  Day of the week the session occurs
</ResponseField>

<ResponseField name="start_time" type="string" required>
  Session start time (format: "HH:MM:SS")
</ResponseField>

<ResponseField name="end_time" type="string" required>
  Session end time (format: "HH:MM:SS")
</ResponseField>

```typescript
type OccurrenceStatus = 'not_started' | 'in_progress' | 'completed'
type Priority = 'low' | 'medium' | 'high'

interface CoverOccurrence {

  meeting_date: string

  actual_end?: string | null
  notes?: string | null
```

<ResponseField name="meeting_date" type="string" required>
  ISO 8601 date string for the scheduled session
</ResponseField>

<ResponseField name="actual_start" type="string | null">
  Actual start time if different from scheduled
</ResponseField>

<ResponseField name="actual_end" type="string | null">
  Actual end time if different from scheduled
</ResponseField>

<ResponseField name="assignments" type="Array<TeacherCoverAssignment>">

### TeacherCoverAssignment

```typescript
type AssignmentStatus = 'invited' | 'accepted' | 'declined' | 'pending' | 'confirmed'

interface TeacherCoverAssignment {
  id: string
  teacher_id: string
  cover_occurrence_id: string
  status: AssignmentStatus
  assigned_by?: string | null
  invited_at?: string | null
  response_at?: string | null
  created_at: string
  teacher?: Teacher
}
```

<ResponseField name="status" type="AssignmentStatus" required>
  Current state of the assignment:
  - `invited`: Teacher has been invited
  - `accepted`: Teacher accepted the assignment
  - `declined`: Teacher declined
  - `pending`: Waiting for response
  - `confirmed`: Assignment is confirmed
</ResponseField>

## Usage Examples

### Creating a New Club

```typescript
import type { Club } from '@/types/club.types'

const newClub: Omit<Club, 'id' | 'created_at' | 'updated_at'> = {
  school_id: 'uuid-here',
  club_name: 'Chess Club',
  club_code: 'CHESS01',
  description: 'Weekly chess strategy and practice',
  members_count: 0,
  status: 'active'
}
```

### Working with Cover Occurrences

```typescript
import type { CoverOccurrence, CoverRule } from '@/types/club.types'

function getOccurrenceTitle(occurrence: CoverOccurrence): string {
  return occurrence.cover_rule?.club?.club_name || 'Unnamed Club'
}

function isUpcoming(occurrence: CoverOccurrence): boolean {
  return new Date(occurrence.meeting_date) > new Date()
}
```

### Type Guards

```typescript
function hasAssignments(occurrence: CoverOccurrence): occurrence is CoverOccurrence & { assignments: Array<TeacherCoverAssignment> } {
  return !!occurrence.assignments && occurrence.assignments.length > 0
}

function isConfirmed(assignment: TeacherCoverAssignment): boolean {
  return assignment.status === 'confirmed' || assignment.status === 'accepted'
}
```
